35       * @param log a logger could be null
36       * @return the <code>JAVA_HOME</code> from System.getProperty( "java.home" )
37       * By default, <code>System.getProperty( "java.home" ) = JRE_HOME</code> and <code>JRE_HOME</code>
38       * should be in the <code>JDK_HOME</code>
39       * @since 2.6
40       */
41      private static File getJavaHome( Log log )
42      {
43          File javaHome;
44          if ( SystemUtils.IS_OS_MAC_OSX )
45          {
46              javaHome = SystemUtils.getJavaHome();
47          }
48          else
49          {
50              javaHome = new File( SystemUtils.getJavaHome(), ".." );
51          }
52  
53          if ( javaHome == null || !javaHome.exists() )
54          {
55              try
56              {
57                  javaHome = new File( CommandLineUtils.getSystemEnvVars().getProperty( "JAVA_HOME" ) );
58              }
59              catch ( IOException e )
60              {
61                  if ( log != null && log.isDebugEnabled() )
62                  {
63                      log.debug( "IOException: " + e.getMessage() );
64                  }
65              }
66          }
67  
68          if ( javaHome == null || !javaHome.exists() )
69          {
70              if ( log != null && log.isErrorEnabled() )
71              {
72                  log.error( "Cannot find Java application directory. Either specify \'java.home\' system property, or "
73                      + "JAVA_HOME environment variable." );
74              }
75          }
76  
77          return javaHome;
78      }

95                  }
96              }
97          }
98          File m2Home = new File( mavenHome );
99          if ( !m2Home.exists() )

101             if ( log != null && log.isErrorEnabled() )

103                 log

105                        + "M2_HOME environment variable." );
106             }
107         }

112 	/**
113      * Invoke Maven for the given project file with a list of goals and properties, the output will be in the
114      * invokerlog file.
115      * <br/>
116      * <b>Note</b>: the Maven Home should be defined in the <code>maven.home</code> Java system property or defined in
117      * <code>M2_HOME</code> system env variables.
118      *
119      * @param log a logger could be null.
120      * @param localRepositoryDir the localRepository not null.
121      * @param projectFile a not null project file.
122      * @param goals a not null goals list.
123      * @param properties the properties for the goals, could be null.
124      * @param invokerLog the log file where the invoker will be written, if null using <code>System.out</code>.
125      * @throws MavenInvocationException if any
126      * @since 2.6
127      */
128     protected static void invokeMaven( Log log, File localRepositoryDir, File projectFile, List<String> goals,
129                                        Properties properties, File invokerLog )
130         throws MavenInvocationException
131     {
132         if ( projectFile == null )
133         {
134             throw new IllegalArgumentException( "projectFile should be not null." );
135         }
136         if ( !projectFile.isFile() )
137         {
138             throw new IllegalArgumentException( projectFile.getAbsolutePath() + " is not a file." );
139         }
140         if ( goals == null || goals.size() == 0 )
141         {
142             throw new IllegalArgumentException( "goals should be not empty." );
143         }
144         if ( localRepositoryDir == null || !localRepositoryDir.isDirectory() )
145         {
146             throw new IllegalArgumentException( "localRepositoryDir '" + localRepositoryDir
147                 + "' should be a directory." );
148         }
149 
150         String mavenHome = getMavenHome( log );
151         if ( StringUtils.isEmpty( mavenHome ) )
152         {
153             String msg =
154                 "Could NOT invoke Maven because no Maven Home is defined. You need to have set the M2_HOME "
155                     + "system env variable or a maven.home Java system properties.";
156             if ( log != null )
157             {
158                 log.error( msg );
159             }
160             else
161             {
162                 System.err.println( msg );
163             }
164             return;
165         }
166 
167         Invoker invoker = new DefaultInvoker();
168         invoker.setMavenHome( new File( mavenHome ) );
169         invoker.setLocalRepositoryDirectory( localRepositoryDir );
170 
171         InvocationRequest request = new DefaultInvocationRequest();
172         request.setBaseDirectory( projectFile.getParentFile() );
173         request.setPomFile( projectFile );
174         if ( log != null )
175         {
176             request.setDebug( log.isDebugEnabled() );
177         }
178         else
179         {
180             request.setDebug( true );
181         }
182         request.setGoals( goals );
183         if ( properties != null )
184         {
185             request.setProperties( properties );
186         }
187         File javaHome = getJavaHome( log );
188         if ( javaHome != null )
189         {
190             request.setJavaHome( javaHome );
191         }
192 
193         if ( log != null && log.isDebugEnabled() )
194         {
195             log.debug( "Invoking Maven for the goals: " + goals + " with "
196                 + ( properties == null ? "no properties" : "properties=" + properties ) );
197         }
198         InvocationResult result = invoke( log, invoker, request, invokerLog, goals, properties, null );
199 
200         if ( result.getExitCode() != 0 )
201         {
202             String invokerLogContent = readFile( invokerLog, "UTF-8" );
203 
204             // see DefaultMaven
205             if ( invokerLogContent != null && ( invokerLogContent.indexOf( "Scanning for projects..." ) == -1
206                 || invokerLogContent.indexOf( OutOfMemoryError.class.getName() ) != -1 ) )
207             {
208                 if ( log != null )
209                 {
210                     log.error( "Error occurred during initialization of VM, trying to use an empty MAVEN_OPTS..." );
211 
212                     if ( log.isDebugEnabled() )
213                     {
214                         log.debug( "Reinvoking Maven for the goals: " + goals + " with an empty MAVEN_OPTS..." );
215                     }
216                 }
217                 result = invoke( log, invoker, request, invokerLog, goals, properties, "" );
218             }
219         }
220 
221         if ( result.getExitCode() != 0 )
222         {
223             String invokerLogContent = readFile( invokerLog, "UTF-8" );
224 
225             // see DefaultMaven
226             if ( invokerLogContent != null && ( invokerLogContent.indexOf( "Scanning for projects..." ) == -1
227                 || invokerLogContent.indexOf( OutOfMemoryError.class.getName() ) != -1 ) )
228             {
229                 throw new MavenInvocationException( ERROR_INIT_VM );
230             }
231 
232             throw new MavenInvocationException( "Error when invoking Maven, consult the invoker log file: "
233                 + invokerLog.getAbsolutePath() );
234         }
235     }

237     /**
238      * @param log could be null
239      * @param invoker not null
240      * @param request not null
241      * @param invokerLog not null
242      * @param goals not null
243      * @param properties could be null
244      * @param mavenOpts could be null
245      * @return the invocation result
246      * @throws MavenInvocationException if any
247      * @since 2.6
248      */
249     private static InvocationResult invoke( Log log, Invoker invoker, InvocationRequest request, File invokerLog,
250                                             List<String> goals, Properties properties, String mavenOpts )
251         throws MavenInvocationException
252     {
253         PrintStream ps;
254         OutputStream os = null;
255         if ( invokerLog != null )
256         {
257             if ( log != null && log.isDebugEnabled() )
258             {
259                 log.debug( "Using " + invokerLog.getAbsolutePath() + " to log the invoker" );
260             }
261 
262             try
263             {
264                 if ( !invokerLog.exists() )
265                 {
266                     invokerLog.getParentFile().mkdirs();
267                 }
268                 os = new FileOutputStream( invokerLog );
269                 ps = new PrintStream( os, true, "UTF-8" );
270             }
271             catch ( FileNotFoundException e )
272             {
273                 if ( log != null && log.isErrorEnabled() )
274                 {
275                     log.error( "FileNotFoundException: " + e.getMessage() + ". Using System.out to log the invoker." );
276                 }
277                 ps = System.out;
278             }
279             catch ( UnsupportedEncodingException e )
280             {
281                 if ( log != null && log.isErrorEnabled() )
282                 {
283                     log.error( "UnsupportedEncodingException: " + e.getMessage()
284                         + ". Using System.out to log the invoker." );
285                 }
286                 ps = System.out;
287             }
288         }
289         else
290         {
291             if ( log != null && log.isDebugEnabled() )
292             {
293                 log.debug( "Using System.out to log the invoker." );
294             }
295 
296             ps = System.out;
297         }
298 
299         if ( mavenOpts != null )
300         {
301             request.setMavenOpts( mavenOpts );
302         }
303 
304         InvocationOutputHandler outputHandler = new PrintStreamHandler( ps, false );
305         request.setOutputHandler( outputHandler );
306 
307         outputHandler.consumeLine( "Invoking Maven for the goals: " + goals + " with "
308             + ( properties == null ? "no properties" : "properties=" + properties ) );
309         outputHandler.consumeLine( "" );
310         outputHandler.consumeLine( "M2_HOME=" + getMavenHome( log ) );
311         outputHandler.consumeLine( "MAVEN_OPTS=" + getMavenOpts( log ) );
312         outputHandler.consumeLine( "JAVA_HOME=" + getJavaHome( log ) );
313         outputHandler.consumeLine( "JAVA_OPTS=" + getJavaOpts( log ) );
314         outputHandler.consumeLine( "" );
315 
316         try
317         {
318             return invoker.execute( request );
319         }
320         finally
321         {
322             IOUtil.close( os );
323             ps = null;
324         }
325     }
